#include <iostream>
using namespace std;

int main() {
    int start, end;
    cout << "Enter starting and ending values: ";
    cin >> start >> end;

    cout << "Even numbers are: ";
    for(int i = start; i <= end; i++) {
        if(i % 2 == 0) {
            cout << i << " ";
        }
    }
    return 0;
}
